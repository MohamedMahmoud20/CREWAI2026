🚀 Professional Flutter Project Structure & Setup Guide (Merged Version)

You are a Flutter AI Assistant working on a multilingual (Arabic + English) mobile app using BLoC for state management, GetX for localization, and structured clean architecture.

Conculation : 
Cursor Rules for Flutter Project

generate this page this in home feature handle it and use homeCubit 
dont make any cubit for any page use home cubit if i in home feature 
use auth cubit if i in auth feature 
use common folder 

1. استخدم هيكلية المجلدات التالية بدقة:
   - /lib/app_manager
   - /lib/auth (cubit, state, models, screens, widgets)
   - /lib/home (cubit, state, models, screens, widgets)
   - /lib/profile (cubit, state, models, screens, widgets)
   - /lib/common
   - /lib/init
   - /lib/services (handle_error.dart, shared_preferences.dart, validators.dart, translations/)
   - /lib/utilities/text_style.dart

2. كل ميزة (feature) يجب أن تحتوي على:
   - cubit/
   - state/
   - models/
   - screens/
   - widgets/

3. اعتمد BLoC (Cubit/Bloc) لإدارة الحالة، ولف كل شاشة بـ BlocProvider و BlocConsumer.

4. استخدم GetX للترجمة (localization) عبر TranslationService، مع ملفات arabic_json.dart و english_json.dart.

5. كل النصوص يجب أن تمر عبر `.tr` للترجمة، ولا تكتب نصوصًا مباشرة في الواجهة.

6. تحقق دائمًا من اتجاه النص (RTL أو LTR) حسب اللغة:
   bool isArabic = Get.locale?.languageCode == 'ar';
   Directionality(
     textDirection: isArabic ? TextDirection.rtl : TextDirection.ltr,
     child: ...
   )

7. اعتمد TextStyle من text_style.dart، واستخدم خط Cairo لجميع النصوص.

8. جميع القياسات يجب أن تكون responsive باستخدام flutter_screenutil (مثل: 14.sp, 20.w, 15.h).

9. استخدم MainButton و MainTextField من common مع تخصيصات واضحة للستايل.

10. جميع عمليات التنقل (navigation) يجب أن تتم حصريًا عبر RoutesManager:
    - RoutesManager.navigatorPush(context, Widget());
    - RoutesManager.navigatorReplace(context, Widget());
    - RoutesManager.navigatorAndRemove(context, Widget());
    - RoutesManager.pop(context);

11. استخدم SharedPreference (مع FlutterSecureStorage) لتخزين واسترجاع البيانات، ووفّر دوال static مثل init, setData, getData.

12. جميع عمليات التحقق (validation) يجب أن تتم عبر Validators في services/validators.dart.

13. عند بناء أي شاشة:
    - استخدم BlocProvider و BlocConsumer
    - تحقق من اللغة واتجاه النص
    - استخدم Form مع GlobalKey
    - اعتمد MainTextField و MainButton
    - اربط كل العمليات بالـ Cubit المناسب

14. جميع النماذج (Models) يجب أن تدعم fromJson و toJson.

15. لا تكرر الكود: اعتمد على widgets وقوالب جاهزة (Reusable Components).

16. لا تترك أي نص أو لون أو ستايل صريح في الكود، بل استخدم الثوابت من الملفات المركزية (text_style.dart, colors.dart).

17. جميع الصور والخطوط يجب أن تُضاف في pubspec.yaml وتكون في مجلد assets.

18. عند إضافة حقل جديد أو شاشة جديدة:
    - أضف الترجمة في ملفات اللغات.
    - أضف الستايل المناسب في text_style.dart إذا لزم.
    - أضف عمليات التحقق في validators.dart إذا لزم.

19. التزم دائمًا بقواعد Clean Architecture وفصل الطبقات (Presentation, Domain, Data).

class RoutesManager {
  static void navigatorPush(context, Widget) {
    Navigator.of(context)
        .push(MaterialPageRoute(builder: (BuildContext context) {
      return Widget;
    }));
  }

  static void pop(context) {
    Navigator.of(context).pop();
  }

  static void navigatorAndRemove(context, Widget) {
    Navigator.of(context).pushAndRemoveUntil(
        MaterialPageRoute(builder: (context) => Widget),
        (Route<dynamic> route) => false);
  }

  static void navigatorReplace(context, Widget) {
    Navigator.of(context)
        .pushReplacement(MaterialPageRoute(builder: (BuildContext context) {
      return Widget;
    }));
  }
}

class SharedPreference {
  static const _storage = FlutterSecureStorage();

  static Future<void> init() async {}

  static Future<void> saveData({required String key, required String value}) async {
    await _storage.write(key: key, value: value);
  }

  static String? getData({required String key}) {
    return _storage.read(key: key) as String?;
  }
}

class OrderModel {
  String? id;
  String? address;
  Status? status;
  Financial? financial;
  Dates? dates;
  String? vendor;
  Delivery? delivery;

  OrderModel({
    this.id,
    this.address,
    this.status,
    this.financial,
    this.dates,
    this.vendor,
    this.delivery,
  });

  OrderModel.fromJson(Map<String, dynamic> json) {
    id = json['id'];
    address = json['address'];
    status = json['status'] != null ? Status.fromJson(json['status']) : null;
    financial = json['financial'] != null ? Financial.fromJson(json['financial']) : null;
    delivery = json['delivery'] != null ? Delivery.fromJson(json['delivery']) : null;
    dates = json['dates'] != null ? Dates.fromJson(json['dates']) : null;
    vendor = json['vendor'];
  }

  Map<String, dynamic> toJson() {
    final Map<String, dynamic> data = {};
    data['id'] = id;
    data['address'] = address;
    if (status != null) data['status'] = status!.toJson();
    if (financial != null) data['financial'] = financial!.toJson();
    if (dates != null) data['dates'] = dates!.toJson();
    data['vendor'] = vendor;
    if (delivery != null) data['delivery'] = delivery!.toJson();
    return data;
  }
}

/// 🧠 الكيوبت الخاص بالطلبات
class OrdersCubit extends Cubit<OrdersStates> {
  OrdersCubit() : super(OrdersInitialState());
  static OrdersCubit get(context) => BlocProvider.of(context);

  Dio dio = Dio();

  void updateToken() {
    dio.options.headers["Authorization"] = "Bearer ${SharedPreference.getData(key: "userToken")}";
    dio.options.headers["Accept"] = "application/json";
    dio.options.headers["x-lang"] = Get.locale?.languageCode == 'ar' ? "ar" : "en";
    dio.options.headers["Content-Type"] = "application/json";
  }

  List<OrderModel> ordersList = [];
  bool isLoadingOrders = false;

  Future<void> getOrders({required BuildContext context}) async {
    try {
      emit(OrdersLoadingState());
      isLoadingOrders = true;
      updateToken();

      var response = await dio.get("https://api.example.com/orders");

      if (response.statusCode == 200) {
        ordersList.clear();
        for (var element in response.data['data']) {
          ordersList.add(OrderModel.fromJson(element));
        }
        emit(OrdersSuccessState());
      }
    } on DioException catch (error) {
      handleError(context: context, error: error);
      emit(OrdersErrorState());
    } finally {
      isLoadingOrders = false;
    }
  }

  Future<void> createOrder({required Map<String, dynamic> data, required BuildContext context}) async {
    try {
      emit(OrderSubmitLoadingState());
      updateToken();

      var response = await dio.post("https://api.example.com/orders", data: data);

      if (response.statusCode == 201) {
        emit(OrderSubmitSuccessState());
      }
    } on DioException catch (error) {
      handleError(context: context, error: error);
      emit(OrderSubmitErrorState());
    }
  }
}

class OrderSubmitSuccessState extends OrdersStates {}

class MainTextField extends StatelessWidget {
  final String? label;
  final int? maxLines;
  final void Function()? onTapEye;
  final TextEditingController? controller;
  final bool? isObscure, isEnabled , showPassword , isPassword;
  final TextInputType? keyboardType;
  final Color? borderColor;
  final Function(String)? onChange , onSubmit;
  final String? Function(String?)? validator;
  final List<TextInputFormatter>? inputFormatters;
  final Widget? suffixIcon, prefixIcon;

  const MainTextField({
    super.key,
    this.label,
    this.controller,
    this.isObscure = false,
    this.isEnabled = true,
    this.keyboardType = TextInputType.text,
    this.onChange,
    this.validator,
    this.inputFormatters,
    this.suffixIcon,
    this.prefixIcon, this.onTapEye, this.showPassword, this.isPassword, this.borderColor, this.maxLines, this.onSubmit,
  });

  @override
  Widget build(BuildContext context) {
    bool isArabic = Get.locale?.languageCode == 'ar';

    return TextFormField(
      controller: controller,
      obscureText: isObscure ?? false,
      enabled: isEnabled ?? true,
      keyboardType: keyboardType,
      inputFormatters: inputFormatters,
      textAlign: isArabic ? TextAlign.right : TextAlign.left,
      validator: validator,
      style: BlackTitle.display5(context).copyWith(fontWeight: FontWeight.w400,fontSize: 15.sp),
      onChanged: onChange,
      onFieldSubmitted: onSubmit,
      maxLines: maxLines??1,
      decoration: InputDecoration(
        hintText:label,
        hintStyle: GreyLabel.display5(context).copyWith(fontWeight: FontWeight.w400,fontSize: 13.sp),
        contentPadding: EdgeInsets.symmetric(horizontal:  14.w,vertical:  14.h), // Adjust padding
        filled: true,
        fillColor: textFieldColor,
        labelStyle: WhiteLabel.display5(context),
        border: OutlineInputBorder(
            borderRadius: BorderRadius.circular(15),
            borderSide: borderColor!=null ? BorderSide(color: borderColor!, width: 1): BorderSide.none),
        enabledBorder: OutlineInputBorder(
            borderSide: borderColor!=null ? BorderSide(color: borderColor!, width: 1): BorderSide.none,
          borderRadius: BorderRadius.circular(15),),
        focusedBorder: OutlineInputBorder(
          borderSide: borderColor!=null ? BorderSide(color: borderColor!, width: 1): BorderSide.none,
          borderRadius: BorderRadius.circular(15),),
        errorBorder: OutlineInputBorder(
            borderSide: BorderSide(color: redColor, width: 1),
            borderRadius: BorderRadius.circular(15)
        ),
        focusedErrorBorder: OutlineInputBorder(
            borderSide: BorderSide(color: redColor, width: 1),
            borderRadius: BorderRadius.circular(15)),

        suffixIcon: suffixIcon ?? (isPassword==true ? InkWell(
          splashColor: transparentColor,
          highlightColor: transparentColor,
          onTap:onTapEye,
          child: Padding(padding:  EdgeInsets.symmetric(horizontal : 15.w),
            child: Stack(
              alignment: Alignment.center,
              children: [
                showPassword==true?  SvgPicture.asset("assets/images/disable.svg",height: 10.h,):SizedBox(),
                SvgPicture.asset("assets/images/eye.svg" , height: 10.h,),
              ],
            ),
          ),
        ):null),
      ),
    );
  }
}

import 'package:get/get.dart';


class Validators {
  static final RegExp _numberRegExp = RegExp(r'^\d+$');
  static final RegExp _nameRegExp = RegExp(r'^[a-zA-Zا-ي\-]+$');

  static String? validateEmail(String? value) {
    if (value == null || value.isEmpty) {
      return 'required_field'.tr;
    }
    if (!GetUtils.isEmail(value)) {
      return 'invalid_email'.tr;
    }
    return null;
  }

  static String? validateName(String? value) {
    if (value == null || value.isEmpty) {
      return 'name_required'.tr;
    } else if (value.length <= 2) {
      return 'name_min_length'.tr;
    } else if (value.length > 10) {
      return 'name_max_length'.tr;
    } else if (!_nameRegExp.hasMatch(value)) {
      return 'name_only_letters'.tr;
    }
    return null;
  }


  static String? validatePassword(String? value) {
    if (value == null || value.isEmpty) {
      return 'required_field'.tr;
    }
    if (value.length < 6) {
      return 'password_too_short'.tr;
    }
    return null;
  }

  static String? validateNumber(String? value) {
    if (value == null || value.isEmpty) {
      return 'number_required'.tr;
    } else if (!_numberRegExp.hasMatch(value)) {
      return 'only_numbers_allowed'.tr;
    } else if (value.length != 8) {
      return 'number_must_be_8_digits'.tr;
    }
    return null;
  }

  static String? validateConfirmPassword(String? value, String? password) {
    if (value == null || value.isEmpty) {
      return 'required_field'.tr;
    }
    if (value != password) {
      return 'passwords_dont_match'.tr;
    }
    return null;
  }

  static String? validateRequired(String? value) {
    if (value == null || value.isEmpty) {
      return 'required_field'.tr;
    }
    return null;
  }
}
